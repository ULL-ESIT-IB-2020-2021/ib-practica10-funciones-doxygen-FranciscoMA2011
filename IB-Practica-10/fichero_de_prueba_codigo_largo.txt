//Comprobar si un correo es correcto
#include <iostream>
#include <string>
#include <cctype>
#include <chrono>
#include <thread>


int main(){
    using namespace std::chrono_literals;
    char character;
    std::string palabra;
    std::cin >> palabra;

    std::cout << "Checking for @. . . ";
    std::this_thread::sleep_for(3s);

    if(!(palabra.find('@'))){
        std::cout << "No tiene un dominio definido." << std::endl;
        std::cout << "Checking failed - Proceeded to exit";
        exit(EXIT_SUCCESS);
    }
    std::cout << "Success!" << std::endl;
    std::this_thread::sleep_for(0.5s);
    int index = palabra.find_first_of('@');
    std::string dominio = palabra.substr(index + 1);
    std::string nombre_mail = palabra.substr(0, index);
    
    std::cout << "Checking for weird characters. . . ";
    std::this_thread::sleep_for(3s);

    for(auto charac: nombre_mail){
        if(! ((charac >= 'a' && charac <= 'z') || (charac >= 'A' && charac <= 'Z') || (charac >= '0' && charac <= '9'))){
            std::cout << "El nombre del mail tiene algun caracter no reconocible." << std::endl;
            std::cout << "Checking failed - Proceeded to exit";
            exit(EXIT_SUCCESS);
        }
    }

    std::cout << "Success!" << std::endl;
    std::this_thread::sleep_for(0.5s);

    std::cout << "Checking for correct domain. . . ";
    std::this_thread::sleep_for(3s);

    if(dominio != "dominiodemierda.com"){
        std::cout << "Dominio incorrecto." << std::endl;
        std::cout << "Checking failed - Proceeded to exit";
        exit(EXIT_SUCCESS);
    }

    std::cout << "Success!" << std::endl;
    std::cout << "Correcto mail";
}
